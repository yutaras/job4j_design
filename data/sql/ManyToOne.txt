create table bus(
    id serial primary key,
    name varchar(255)
);

create table driver(
    id serial primary key,
    name varchar(255),
    bus_id int references bus(id)
);

insert into bus(name) values ('MAN');
insert into driver(name, bus_id) VALUES ('Ivan', 1);

select * from driver;

select * from bus where id in (select id from driver);
